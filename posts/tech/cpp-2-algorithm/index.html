1.高精度算法

是指运算的结果超过数据类型本身的范围.

加法

 1#include <iostream>
 2using namespace std;
 3
 4
 5//存放输入数据以及结果
 6int a[101], b[101], c[101];
 7string s1, s2;
 8
 9int main() {
10	cin >> s1 >> s2;
11	//1.个位反转
12	strtoint(s1, a);
13	strtoint(s2, b);
14	int la = s1.size(), lb = s2.size();
15	
16	//2.计算c数组的长度
17	int lc = max(la, lb) + 1; //考虑进位
18
19	//3.对位相加得到c数组
20	for (int i = 1; i <= lc; i++) {
21		c[i] = a[i] + b[i] + c[i];
22		c[i + 1] = c[i] / 10; //进位
23		c[i] %= 10; //留个位
24	}
25
26	//4.去除高位0
27	while (c[lc] == 0 && lc > 1)lc--;
28
29	//5.打印
30	for (int i = lc; i >= 1; i--) {
31		cout << c[i];
32	}
33
34	//cout << "Hello World" << endl;
35	return 0;
36}
37
38void strtoint(string src, int des[]) {
39	for (int i = 0; i < src.size(); i++) {
40		des[src.size() - i] = src[i] - '0';
41	}
42}

减法

 1#include <iostream>
 2using namespace std;
 3
 4
 5//存放输入数据以及结果
 6int a[101], b[101], c[101];
 7string s1, s2;
 8
 9int main() {
10	cin >> s1 >> s2;
11	//判断结果是否为负
12	if (cmpstr(s1, s2) == false) {
13		swap(s1, s2);
14		cout << "-";
15	}
16	//1.转数组
17	strtoint(s1, a);
18	strtoint(s2, b);
19	int la = s1.size(), lb = s2.size();
20	int lc = max(la, lb); 
21
22	//2.进行减法
23	for (int i = 1; i <= lc; i++) {
24		//不够减, 借位
25		if (a[i] < b[i]) {
26			a[i + 1]--;
27			a[i] += 10;
28		}
29		c[i] = a[i] - b[i];
30	}
31
32	//3.去零
33	while (c[lc] == 0 && lc > 1) lc--;
34	
35	//4.输出
36	for (int i = lc; i >= 1; i--)
37		cout << c[i];
38
39	return 0;
40}

15

18
19	//3.计算乘法
20	for (int i = 1; i <= la; i++) {
21		for (int j = 1; j <= lb; j++) {
22			c[i + j - 1] += a[i] * b[j];
23			c[i + j] += c[i + j - 1] / 10;
24			c[i + j - 1] %= 10;
25		}
26	}
27
28	//4.去除高位0
29	while (c[lc] == 0 && lc > 1)lc--;
30
31	//5.打印

33		cout << c[i];
34
35	return 0;

37}
